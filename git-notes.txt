# Basic git commands
git init
git status
git add ...
git rm ...
git ignore ...
git commit -m "..."

# Github connection
ssh -T -p 443 git@ssh.github.com
ssh -T  git@ssh.github.com

# Define a remote
git remote add origin git@github.com:FabriceSalvaire/PyElectronic.git
git remote del origin
git remote -v

# Clone a remote
git clone git://github.com/FabriceSalvaire/PyElectronic.git

# Push commits to a remote
git push -u origin master
git push -u origin gh-pages

git fetch origin
git checkout gh-pages
